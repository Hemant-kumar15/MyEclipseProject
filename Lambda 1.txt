interface LightAction {
    void activate();
}

public class SmartHomeLighting {
    public static void main(String[] args) {
        LightAction motion = () -> System.out.println("Motion detected → Lights ON!");
        LightAction night = () -> System.out.println("Night time → Dim warm lights ON!");
        LightAction voice = () -> System.out.println("Voice command → Living room lights ON!");
        motion.activate();
        night.activate();
        voice.activate();
    }
}
