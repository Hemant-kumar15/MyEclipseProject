interface Department {
    void assignDepartment(String d);
    String getDepartmentDetails();
}

abstract class Employee {
    private int id; private String name; private double base;
    public Employee(int id,String n,double b){this.id=id;this.name=n;this.base=b;}
    public double getBase(){return base;}
    abstract double calculateSalary();
    void display(){System.out.println(id+" "+name+" "+base);}
}

class FullTimeEmployee extends Employee implements Department {
    String dept;
    FullTimeEmployee(int i,String n,double b){super(i,n,b);}
    double calculateSalary(){return getBase()+5000;}
    public void assignDepartment(String d){dept=d;}
    public String getDepartmentDetails(){return dept;}
}

class PartTimeEmployee extends Employee implements Department {
    int hrs; String dept;
    PartTimeEmployee(int i,String n,double b,int h){super(i,n,b);hrs=h;}
    double calculateSalary(){return getBase()*hrs;}
    public void assignDepartment(String d){dept=d;}
    public String getDepartmentDetails(){return dept;}
}

public class Main {
    public static void main(String[] a){
        Employee e1=new FullTimeEmployee(1,"Amit",30000);
        Employee e2=new PartTimeEmployee(2,"Ravi",500,40);
        ((Department)e1).assignDepartment("HR");
        ((Department)e2).assignDepartment("Support");
        for(Employee e:new Employee[]{e1,e2}){
            e.display();
            System.out.println("Dept:"+((Department)e).getDepartmentDetails());
            System.out.println("Salary:"+e.calculateSalary()+"\n");
        }
    }
}
